## External Validation Command

Run on the VPS:

```bash
cd /root/veloce-research-os
git pull --ff-only

echo "1. Repo commit"
git rev-parse --short HEAD

echo
echo "2. Reliability policy matrix tests"
cd /root/veloce-research-os/projects/reliability-policy-matrix
python3 -m unittest discover -s tests

echo
echo "3. Matrix plan"
make matrix-plan | sed -n '1,80p'

echo
echo "4. Sample run"
make run-sample

echo
echo "5. Aggregate sample"
make aggregate-sample

echo
echo "6. Artifact listing"
find artifacts -maxdepth 3 -type f | sort

echo
echo "7. MCPO Ruflo status"
cd /root/ai-agency
set -a
source .env
set +a
curl -sS https://tools.srv1314350.hstgr.cloud/ruflo_status \
  -H "Authorization: Bearer $MCPO_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"scope":"sandbox"}' | python3 -m json.tool

echo
echo "8. MCPO Ruflo execution packet"
curl -sS https://tools.srv1314350.hstgr.cloud/ruflo_execution_packet \
  -H "Authorization: Bearer $MCPO_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"issue_id":"VEL-129","title":"v1.7A production-readiness audit","description":"External VPS verification for the production-readiness audit after Paperclip runtime blocked Python execution.","approved_by":"Aryaman","approval":"human_approved","execution_owner":"Codex/GitHub/VPS","labels":["v1.7","production-readiness"],"requested_mode":"execution_packet"}' | python3 -m json.tool

echo
echo "9. Proxy log proof"
docker logs --tail=80 aiagency-mcpo-openwebui-proxy | grep -E 'ruflo_status|ruflo_execution_packet'
```
